# useful for handling different item types with a single interface
from itemadapter import ItemAdapter


class HousinganywhereProjectPipeline:

    def process_item(self, item, spider):
        
        adapter = ItemAdapter(item)

        ## Strip all whitespaces from strings
        '''
        Con este adaptador podemos obtener todos los nombres de los campos
        Y le decimos que si es precio, queremos quedarnos con el primer valor
        '''

        field_names = adapter.field_names()
        for field_name in field_names:
            if field_name == 'price':
                value = adapter.get(field_name)
                if value is not None:
                    adapter[field_name] = int(value.split()[0])
                else:
                    adapter[field_name] = None 
        
        return item
